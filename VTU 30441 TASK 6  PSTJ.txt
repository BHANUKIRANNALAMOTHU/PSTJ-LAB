import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        int[] arr = {7, 2, 5, 1, 9};

        Arrays.sort(arr);

        Scanner sc = new Scanner(System.in);
        System.out.print("Enter K: ");
        int k = sc.nextInt();

        System.out.println(k + "th Smallest = " + arr[k - 1]);
    }
}

OUTPUT:
Enter K: 5
5th Smallest = 9
