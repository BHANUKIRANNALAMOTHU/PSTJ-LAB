import java.util.function.Function;

public class Main {
    public static void main(String[] args) {
        Function<Integer, Integer> fib = n -> {
            if (n <= 1) return n;

            int a = 0, b = 1;
            for (int i = 2; i <= n; i++) {
                int c = a + b;
                a = b;
                b = c;
            }
            return b;
        };

        System.out.println(fib.apply(7));
    }
}


OUTPUT:
13
